### 倍数计算逻辑

#### 稳健动态投注
```python
def calculate_stable_dynamic_bet(consecutive_wins, consecutive_losses):
    """稳健动态投注 - 连胜保持1倍，连败快速加倍"""
    base_multiplier = 1.0
    
    if consecutive_wins > 0:
        # 连胜期间保持1倍，不减少投注
        multiplier = base_multiplier
    elif consecutive_losses == 1:
        # 连败1期：2倍
        multiplier = 2.0
    elif consecutive_losses == 2:
        # 连败2期：4倍
        multiplier = 4.0
    else:
        # 连败3期+：每期再+2倍
        multiplier = 4.0 + (consecutive_losses - 2) * 2
    
    # 最大10倍限制
    multiplier = min(multiplier, 10.0)
    bet_amount = multiplier * base_bet  # base_bet = 17元
    
    return multiplier, bet_amount
```

#### 选择性动态投注
```python
def calculate_selective_dynamic_bet(consecutive_wins, consecutive_losses):
    """选择性动态投注 - 需要连续2期才触发加倍/减倍"""
    base_multiplier = 1.0
    
    if consecutive_wins >= 3:
        multiplier = 0.5  # 连胜3期+：保守0.5倍
    elif consecutive_wins == 2:
        multiplier = 0.8  # 连胜2期：渐进降低
    elif consecutive_losses >= 2:
        if consecutive_losses == 2:
            multiplier = 2.0  # 连败2期：开始加倍
        elif consecutive_losses == 3:
            multiplier = 4.0
        else:
            multiplier = 4.0 + (consecutive_losses - 3) * 2
    else:
        multiplier = base_multiplier  # 连胜1期或连败1期：保持1倍
    
    multiplier = min(multiplier, 10.0)
    bet_amount = multiplier * base_bet
    should_bet = True  # 总是投注（不跳过）
    
    return multiplier, bet_amount, should_bet
```
